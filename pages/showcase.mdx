# Nodes Showcase

import { Card, Cards } from 'nextra/components';
import showcase from './showcase.json';

<Cards>
  {showcase.map((group) => (
    <Card key={group.uid} image arrow title={group.name} href={`/showcase/${group.uid}`}>
      <img src={group.iconUrl} alt={group.name} />
    </Card>
  ))}
</Cards>

export const ShowcaseCard = Object.assign(
  // Copy card component and add default props
  Card.bind(),
  {
    displayName: 'ShowcaseCard',
    defaultProps: {
      image: true,
      arrow: true,
      target: '_self',
    },
  },
);
